# SSH portu default olan 22'den 2299'a ayarlandı.
Port 2299

# Saldırganların root olarak yapabileceği daha kolay saldırıları engellemek için root girişi engellendi
PermitRootLogin no

# Parola doğrulaması etkinleştirildi.
PasswordAuthentication yes

# Kaliden giriş için key oluşturulduğundan Public Key doğrulaması aktifleştirildi.
PubkeyAuthentication yes
